ѕри запуске данного кода в консоль выведет "Bad" 4 раза. ѕричиной тому переменна€ var, 
у которой нет блочной области видимости и к тому моменту, когда асинхронный setTimeout отработает,  у переменной var будет значение 4.

„тобы этого избежать, можно либо заменить var на let:
for (let i = 0; i < arr.length; i++) {
  setTimeout(function() {
    console.log(arr[i] > 13 ? `Good ${arr[i]}` : `Bad ${arr[i]}`)
  }, 3000);
};

Ћибо передать в коллбек функции setTimeout текущее значение i:
for (var i = 0; i < arr.length; i++) {
  setTimeout(function(i) {
    console.log(arr[i] > 13 ? `Good ${arr[i]}` : `Bad ${arr[i]}`)
  }, 3000, i);
};
